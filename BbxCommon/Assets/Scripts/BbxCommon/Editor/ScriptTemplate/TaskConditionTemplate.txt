using System;
using System.Collections.Generic;
using BbxCommon;

public class #SCRIPT_NAME# : TaskConditionBase
{
    public int IntValue;

    public enum EField
    {
        IntValue,
    }

    protected override Type GetFieldEnumType()
    {
        return typeof(EField);
    }

    protected override void RegisterField()
    {
        RegisterField(EField.IntValue, IntValue);
    }

    public override void ReadFieldInfo(int fieldEnum, TaskFieldInfo fieldInfo, TaskContextBase context)
    {
        switch (fieldEnum)
        {
            case (int)EField.IntValue:
                IntValue = ReadInt(fieldInfo, context);
                break;
        }
    }

    public override void OnCollect()
    {
        IntValue = 0;
    }

    protected override EConditionState OnConditionUpdate(float deltaTime)
    {
        return EConditionState.Succeeded;
    }
}
