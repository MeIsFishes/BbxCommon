{

    {
        IntValue,
    }

    }

    protected override void OnTaskCollect()
    {
        IntValue = 0;
    }

    protected override void OnEnter()
    {
        
    }

    protected override ETaskRunState OnUpdate(float deltaTime)
    {
        return ETaskRunState.Succeeded;
    }

    protected override void OnExit()
    {
        
    }
}
